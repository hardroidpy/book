
\section{Conclusión}
\begin{frame}{Conclusión}

\framesubtitle{Logros}

\setbeamercovered{transparent}
\begin{itemize}
\item Entrenamiento de un clasificador \structure{HAR}. 
\begin{itemize}
\item Aciertos del 91\%
\item Confiabilidad de datos del 87\%.
\item Errores absolutos menores a 1\%. 
\item Precisión y exhaustividad entre el 91\% y el 92\%.
\end{itemize}

\pause{}
\item Construcción de \structure{HARDroid} para teléfonos inteligentes. 
\begin{itemize}
\item Reutilizable y extensible que clasifica diversas actividades.
\item Se mejora el clasificador mediante la inclusión de muestras por colaboración.
\end{itemize}

\pause{}
\item \structure{HARDroid} está disponible para otras aplicaciones móviles:
\begin{itemize}
\item incluido en tiempo de construcción, o
\item como servicio a través de una interfaz de integración.
\end{itemize}
\end{itemize}
\end{frame}
