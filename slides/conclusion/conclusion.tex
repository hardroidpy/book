
\section{Conclusión}
\begin{frame}{Conclusión}

\framesubtitle{Aportes}

\setbeamercovered{transparent}
\begin{itemize}
\item Construcción de \structure{HARDroid} para dispositivos inteligentes. 
\begin{itemize}
\item Reutilizable y extensible que clasifica diversas actividades.
\item Se mejora el clasificador mediante colaboración.
\end{itemize}
\begin{spacing}{0.5}

\pause{}
\end{spacing}
\item \structure{HARDroid} está disponible para otras aplicaciones móviles:
\begin{itemize}
\item incluido en tiempo de construcción, o
\item como servicio a través de una interfaz de integración.
\end{itemize}
\begin{spacing}{0.5}

\pause{}
\end{spacing}
\item Primera librería de reconocimiento \structure{open-source} para \emph{Android}
\begin{itemize}
\item Aporte al estado del arte de las \structure{HAR} con software libre
para el estudio en dispositivos móviles.
\item Implementación alternativa a \emph{Google Play Services} (\emph{closed source}).
\end{itemize}
\end{itemize}
\end{frame}
