\begin{frame}{Metodología HAR}
\framesubtitle{Proceso de aprendizaje}
\begin{itemize}
	\item Como en otros métodos de aprendizaje automático, el proceso de reconocimiento se divide en dos etapas conocidas:
	\begin{itemize}
		\item Entrenamiento
		\item Evaluación
	\end{itemize}
	\item En el entrenamiento requiere:
		\begin{itemize}
			\item Colecta de conjunto de Datos en series de tiempo.
			\item Se realiza la extracción de muestras en ventanas de tiempo, calculando las características de entrenamiento.
			\item Se genera un modelo de Reconocimiento de Actividades a través de un método de aprendizaje automático.
		\end{itemize}
	\item En la evaluación, de igual manera, se recopila los datos, se realiza la extracción de muestras y calculo de las mismas características del entrenamiento, de modo a evaluar el modelo generado.
\end{itemize}
\end{frame}
%
\begin{frame}{Metodología HAR}
\framesubtitle{Colección de Datos}
	\begin{itemize}
		\item El primer paso es recolectar señales obtenidas de sensores unidos al cuerpo (cintura, muñeca, muslos, etc).
		
		\item El método de grabación consiste en capturar las señales de un sensor o más sensores a un individuo realizando alguna actividad de interés.
		
		\item Las señales de sensores pueden clasificarse según el movimiento, la posición, el entorno y la fisiología.	
	\end{itemize}
\end{frame}

\begin{frame}{Metodología HAR}
\framesubtitle{Extracción de Muestras}
	\begin{itemize}
		\item El siguiente paso es el procesamiento de las señales obtenidas por los sensores y extraer las características relevantes de los datos sin procesar.
		\item El procesamiento de muestra se basa en cuatro tareas distintas que se realizan automáticamente en ambas etapas del proceso HAR, las cuales son:
		\begin{itemize}
			\item Etiquetado.
			\item Suavizado.
			\item Muestreo.
			\item Extracción de Características.
		\end{itemize}
	\end{itemize}
\end{frame}
%
\begin{frame}{Metodología HAR}
\framesubtitle{Extracción de Muestras}
\begin{columns}
	
\column{6cm}
	\begin{itemize}[<+->]
		\item \underline{Etiquetado}: Los datos de los sensores se deben recopilar y etiquetar utilizando una aplicación diseñada para el caso, por ejemplo, en este trabajo se usa un teléfono Android y la aplicación SensorLog.
		\item \underline{Suavizado}: Los sensores electrónicos pueden introducir cierta inestabilidad en la señal (jitter) que afectan la calidad de los datos. Para reducir el ruido de la señal, se deben aplicar uno o más filtros. 
	\end{itemize}

\begin{frame}{Metodología HAR}
\framesubtitle{Extracción de Muestras - Muestreo}
\begin{columns}
	
\column{6cm}
	\begin{itemize}[<+->]
		\item \underline{Muestreo}: Los datos colectados se segmentan en ventanas de tiempo de muestras discretas utilizadas como unidades de reconocimiento de actividad. Para aumentar el número de muestras, se utiliza una superposición del 50\%.
		\item \underline{Extracción de Características}: El proceso de extracción consiste en extraer valores característicos en cada ventana de tiempo mediante la visualización de información relevante y el cálculo de valores que identifican de cierta manera las señales.
	\end{itemize}

\begin{frame}{Metodología HAR}	
\framesubtitle{Aprendizaje e inferencia}
\begin{itemize}
	\item Un HAR es similar a cualquier aplicación de aprendizaje automático donde se requiere un algoritmo para extraer información de los datos. El objetivo principal del algoritmo es clasificar los datos desconocidos.
	\item Hay muchos algoritmos de clasificación, como Arboles de Decisión, SVM, clasificadores de Bayes, Modelos de Markov, Redes Neuronales y otros.
\end{itemize}
\end{frame}

\begin{frame}
{Metodología HAR}	
\framesubtitle{Árbol de Decisión}
\begin{columns}
	
\column{6cm}
	\begin{itemize}[<+->]
		\item Método de aprendizaje predictivo sobre un conjunto de tuplas o instancias etiquetadas.
		\item Cada nodo interno (nodo no hoja) denota una prueba de un atributo, y cada nodo hoja es una etiqueta o clase.
		\item Utiliza el criterio de proporción de ganancia.
		\item Simple de Implementar.
		\item Algoritmos de J. R. Quinlan ID3, C4.5(Implementación Java j48).
	\end{itemize}
