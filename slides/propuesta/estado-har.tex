
\section{Metodología HAR}

\begin{frame}[t]{Metodología HAR}
\framesubtitle{Definición Teórica}

\textbf{[Problema HAR]} \\~\

Dado un conjunto $W=\{w_{0},...,w_{m-1}\}$ con $m$ ventanas de tiempo \\~\ 

donde cada $w_{j}$ contiene series de tiempo $S_{j}=\{s_{j,0},...,s_{j,k-1}\}$ para $k$ señales \\~\ \par

Y un conjunto $A\text{=}\{\mathrm{a}_{0},...,\mathrm{a}_{n-1}\}$ de actividades \\~\

El objetivo es encontrar $\mathbf{  F\colon S_{j}\rightarrow A }$ para todos los $S_{j}$ \\~\

\end{frame}

\begin{frame}{Metodología HAR}
\framesubtitle{Extracción de Muestras}
	\begin{itemize}
		\setlength\itemsep{1em}
		\item El siguiente paso es el procesamiento de las señales obtenidas por los sensores y extraer las características relevantes de los datos sin procesar.
		\item El procesamiento de muestra se basa en cuatro tareas distintas:
		\begin{itemize}
			\item Etiquetado.
			\item Suavizado.
			\item Muestreo.
			\item Extracción de Características.
		\end{itemize}
	\end{itemize}
\end{frame}

		\item \underline{Suavizado}: Los sensores electrónicos pueden introducir cierta inestabilidad en la señal (jitter) para reducir esto se aplica uno o mas filtros.

\begin{frame}{Metodología HAR}	
\framesubtitle{Aprendizaje e inferencia}
\begin{itemize}
	\setlength\itemsep{1em}
	\item Un HAR es similar a cualquier aplicación de aprendizaje automático. 
	\item El objetivo principal del algoritmo es clasificar los datos desconocidos.
	\item Hay muchos algoritmos de clasificación: 
	\begin{itemize}
		\item Arboles de Decisión
		\item SVM
		\item Clasificadores de Bayes, 
		\item Modelos de Markov
		\item Redes Neuronales y otros.
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
{Metodología HAR}	
\framesubtitle{Árbol de Decisión}
\begin{columns}
	
\column{6cm}
	\begin{itemize}[<+->]
		\item Método de aprendizaje predictivo sobre un conjunto de tuplas o instancias etiquetadas.
		\item Cada nodo interno (nodo no hoja) denota una prueba de un atributo, y cada nodo hoja es una etiqueta o clase.
		\item Utiliza el criterio de proporción de ganancia.
		\item Simple de Implementar.
		\item Algoritmos de J. R. Quinlan ID3, C4.5(Implementación Java j48).
	\end{itemize}
