
\section{Introducción}

\subsection{Definición del Problema}
\begin{frame}{Definición del Problema}

\framesubtitle{Reconocimiento de Actividades Humanas}
\begin{quote}
Determinar las acciones o comportamientos de uno o más individuos
a partir de datos ambiguos capturados por sensores situados en el
entorno.
\end{quote}
\begin{block}<2>{Nota}
\emph{}El problema es conocido por sus siglas \structure{HAR} en
inglés \emph{(Human Activity Recognition})
\end{block}
\end{frame}
%
\begin{frame}{Definición del Problema}

\setbeamercovered{transparent}

\framesubtitle{¿Que es HAR?}
\begin{itemize}[<+->]
\item Es un tópico de investigación multidisciplinario que busca diseñar
algoritmos para:
\begin{itemize}
\item Capturar movimientos de uno o más individuos en interacción con su
entorno 
\item Realizar el aprendizaje e inferencia
\item Proveer información de contexto 
\end{itemize}
\end{itemize}

\begin{frame}{Definición del Problema}

\framesubtitle{Motivación}
\begin{columns}

\column{6cm}
\begin{itemize}[<+->]
\item Reconocer el \structure{contexto} es primordial en sistemas inteligentes 
\item Avances acelerados en tecnologías de \structure{computación móvil}
y \structure{sensores} 
\item Uso intensivo de \structure{teléfonos móviles modernos} por personas
en su vida diaria.
\item \setbeamercovered{transparent}Popularidad de \structure{aplicaciones móviles}
de contexto en ámbito de
\begin{itemize}
\item cuidado personal, 
\item la movilidad y 
\item la asistencia en la vida diaria 
\end{itemize}
\end{itemize}
