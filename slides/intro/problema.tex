\framesubtitle{¿Que es HAR?}
\begin{itemize}[<+->]
\item Es un tópico de investigación multidisciplinario que busca diseñar
algoritmos para:
\begin{itemize}
\item Capturar movimientos de uno o más individuos en interacción con su
entorno 
\item Realizar aprendizaje e inferencia
\item Proveer información de contexto 
\end{itemize}
\end{itemize}

\begin{frame}{Definición del Problema}

\framesubtitle{Definición Teorica}

\begin{definition}[Problema HAR relajado]Dado (1) un conjunto $W=\{w_{0},...,w_{m-1}\}$
de $m$ ventanas de tiempo del mismo tamaño, donde cada una está total
o parcialmente etiquetada, y que cada $w_{j}$ contiene un conjunto
de series de tiempo $S_{j}=\{s_{j,0},...,s_{j,k-1}\}$ para cada $k$
atributos medidos, y (2) un conjunto $A\text{=}\{\mathrm{a}_{0},...,\mathrm{a}_{n-1}\}$
de etiquetas de actividades, el objetivo es encontrar una función
$f\colon S_{j}\rightarrow A$ que sea evaluada para todos los valores
posibles de $S_{j}$, tal que $f(S_{j})$ es lo más próximo a la acción
realizada durante $w_{j}$ \end{definition}
\end{frame}
