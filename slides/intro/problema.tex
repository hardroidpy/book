\subsection{Definición del Problema}
\begin{frame}{Definición del Problema}

\framesubtitle{Reconocimiento de Actividades Humanas}
\begin{overprint}
\onslide<1> 
\begin{quote}
Determinar las acciones o comportamientos de uno o más individuos
a partir de datos ambiguos capturados por sensores situados en el
entorno.
\end{quote}
\onslide<2> 
\begin{block}<2>{Nota}
\emph{}El problema es conocido por sus siglas \structure{HAR} en
inglés \emph{(Human Activity Recognition})
\end{block}

\begin{frame}{Definición del Problema}

\setbeamercovered{transparent}

\framesubtitle{¿Que es HAR?}
\begin{itemize}[<+->]
\item Es un tópico de investigación multidisciplinario que busca diseñar
algoritmos para:
\begin{itemize}
\item Capturar movimientos de uno o más individuos en interacción con su
entorno 
\item Realizar aprendizaje e inferencia
\item Proveer información de contexto 
\end{itemize}
\end{itemize}

\begin{frame}[shrink]{Definición del Problema}

\framesubtitle{Actividades Humanas}

\setbeamercovered{transparent}
\begin{block}{Actividades Simples}
Acciones simples periódicas de larga duración.
\end{block}
\begin{center}
\uncover<2->{\begin{center}
\begin{tabular}{|c|>{\raggedright}p{5cm}|}
\hline 
Ambulatorias & Caminar, correr, sentarse, pararse, quieto, acostarse, subir y descender
escaleras.\tabularnewline
\hline 
Transporte & En bus, bicicleta y conducir.\tabularnewline
\hline 
\end{tabular}
\par\end{center}}
\par\end{center}
\begin{block}{Actividades Complejas}
Acciones cortas o periódicas no comunes.
\end{block}
\begin{center}
\uncover<3->{\begin{center}
\begin{tabular}{|c|>{\raggedright}p{5cm}|}
\hline 
Cotidianas & Comer, beber, mirar TV, leer, cepillarse los dientes, entre otros.\tabularnewline
\hline 
Gimnasio & Alzar pesas, ejercicio aeróbico, bicicleta estática.\tabularnewline
\hline 
Militares & Arrastrarse, en cuclillas.\tabularnewline
\hline 
\end{tabular}
\par\end{center}}
\par\end{center}
